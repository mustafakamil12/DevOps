from confluent_kafka import Producer

# Kafka broker details
bootstrap_servers = "your_kafka_broker:9093"  # Replace with your broker address

# Path to your SSL certificates
ssl_certificate = "/path/to/certificate.cer"
ssl_private_key = "/path/to/privatekey.key"

# Kafka configuration
config = {
    "bootstrap.servers": bootstrap_servers,
    "security.protocol": "SSL",
    "ssl.certificate.location": ssl_certificate,
    "ssl.key.location": ssl_private_key,
}

# Create a Kafka producer
producer = Producer(config)

# Kafka topic
topic = "your_topic_name"

def delivery_report(err, msg):
    """Callback for delivery reports."""
    if err:
        print(f"Message delivery failed: {err}")
    else:
        print(f"Message delivered to {msg.topic()} [{msg.partition()}] at offset {msg.offset()}")

# Produce messages
for i in range(10):
    message = f"Hello Kafka! Message {i}"
    producer.produce(topic, key=str(i), value=message, callback=delivery_report)

# Wait for all messages to be delivered
producer.flush()
