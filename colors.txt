# Convert JKS to PKCS12 format
keytool -importkeystore -srckeystore my_cert.jks -destkeystore my_cert.p12 -deststoretype PKCS12

# Extract private key
openssl pkcs12 -in my_cert.p12 -nocerts -nodes -out client-key.pem

# Extract certificate
openssl pkcs12 -in my_cert.p12 -clcerts -nokeys -out client-cert.pem

# Extract CA certificate (if needed)
openssl pkcs12 -in my_cert.p12 -cacerts -nokeys -out ca-cert.pem



conf = {
    "bootstrap.servers": "your_kafka_broker:9093",
    "security.protocol": "SSL",
    "ssl.key.location": "client-key.pem",
    "ssl.certificate.location": "client-cert.pem",
    "ssl.ca.location": "ca-cert.pem"
}
